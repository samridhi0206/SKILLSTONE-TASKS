var express = require('express');
var app = express();

var print = require('./myFirstModule');

app.use(function (req, res) { 
  res.send(print.text()); 
});

app.listen(3000, function () {
    console.log('Example app listening on port 3000.');
});

//myFirstModule.js
exports.text = function () {
    return "Hello world";
  };
